✅ Ejercicio 2

// Pipeline que une inscripciones_refer con estudiantes y materias
db.inscripciones_refer.aggregate([
  {
    $lookup: {
      from: "estudiantes",
      localField: "estudiante_id",
      foreignField: "_id",
      as: "estudiante"
    }
  },
  { $unwind: "$estudiante" },
  {
    $lookup: {
      from: "materias",
      localField: "materia_id",
      foreignField: "_id",
      as: "materia"
    }
  },
  { $unwind: "$materia" },
  {
    $project: {
      _id: 0,
      nombre_estudiante: "$estudiante.nombre",
      nombre_materia: "$materia.nombre",
      fecha: 1
    }
  }
]);

✅ Ejercicio 3

// Pipeline que agrupa inscripciones por materia, cuenta y ordena
db.inscripciones.aggregate([
  {
    $group: {
      _id: "$materia",
      total_inscripciones: { $sum: 1 }
    }
  },
  { $sort: { total_inscripciones: -1 } },
  { $limit: 5 }
]);
